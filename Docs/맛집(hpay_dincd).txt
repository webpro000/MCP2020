CREATE TABLE public."tbl_dincd_info"
(
    business_id integer NOT NULL,
    name character varying(100) NOT NULL,
    address character varying(500),
    telephone character varying(50),
    longitude real,
    latitude real,
    avg_rating real,
    ai_score real,
    open_hour character varying(200),
    rep_photo character varying(2000),
    menu_photo character varying(2000),
    area character varying(50),
    r_like_count integer,
    rep_food_keyword character varying(300),
    rep_keyword character varying(300),
    taste_rating real,
    price_rating real,
    service_rating real,
    visit_purpose character varying(200),
    facility character varying(200),
    atmosphere character varying(200),
    regdt timestamp with time zone default now(),
    uptdt timestamp with time zone,
    PRIMARY KEY (business_id)
)

    OIDS = FALSE
